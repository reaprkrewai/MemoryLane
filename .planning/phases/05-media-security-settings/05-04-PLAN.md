### 4.1: Data Collection Logic

**File:** `src/hooks/useDataExport.ts` (new)

Create a hook that collects all data for export:

```typescript
interface ExportData {
  version: string;
  exportedAt: string; // ISO 8601
  entries: Array<{
    id: string;
    date: string;
    content: string;
    mood?: string;
    tags: Array<{ id: string; name: string; color: string }>;
    photos: Array<{ path: string; filename: string }>; // Relative paths in ZIP
  }>;
  tags: Array<{
    id: string;
    name: string;
    color: string;
  }>;
  settings: {
    theme: 'light' | 'dark';
    fontSize: 'small' | 'medium' | 'large';
    idleTimeout: number | 'never';
    appVersion: string;
    exportDate: string;
  };
}

const useDataExport = () => {
  const collectExportData = async (): Promise<ExportData> => {
    const allEntries = await entryStore.getAllEntries();
    const allTags = await tagStore.getAllTags();
    const { theme, fontSize, idleTimeout } = useUiStore.getState();

    // Map entries to export format
    const entries = allEntries.map((entry) => ({
      id: entry.id,
      date: new Date(entry.date).toISOString(),
      content: entry.content,
      mood: entry.mood,
      tags: entry.tags.map((t) => ({
        id: t.id,
        name: t.name,
        color: t.color,
      })),
      photos: (entry.photos || []).map((p) => ({
        path: `photos/${entry.id}/${path.basename(p.photoPath)}`,
        filename: path.basename(p.photoPath),
      })),
    }));

    return {
      version: '1.0.0',
      exportedAt: new Date().toISOString(),
      entries,
      tags: allTags.map((t) => ({ id: t.id, name: t.name, color: t.color })),
      settings: {
        theme,
        fontSize,
        idleTimeout: idleTimeout === 'never' ? -1 : idleTimeout,
        appVersion: import.meta.env.VITE_APP_VERSION || '1.0.0',
        exportDate: new Date().toISOString(),
      },
    };
  };

  return { collectExportData };
};
```

**Acceptance Criteria:**
- [ ] Data collection includes all entries, tags, photos
- [ ] Data structure matches export spec
- [ ] Timestamps are ISO 8601 format
- [ ] Photo paths are relative (for ZIP structure)
- [ ] Settings are included

### 4.2: ZIP File Creation

**File:** `src/utils/zipUtils.ts` (new)

Create utility to build ZIP file from export data:

```typescript
import JSZip from 'jszip'; // or native Tauri lib

interface ZipExportOptions {
  data: ExportData;
  photoBasePath: string; // App data directory path
}

export const createExportZip = async (options: ZipExportOptions): Promise<Blob> => {
  const { data, photoBasePath } = options;
  const zip = new JSZip();

  // Add metadata.json
  const metadataJson = JSON.stringify(data, null, 2);
  zip.file('metadata.json', metadataJson);

  // Add photos
  for (const entry of data.entries) {
    if (entry.photos && entry.photos.length > 0) {
      for (const photo of entry.photos) {
        const photoPath = path.join(photoBasePath, 'photos', entry.id, photo.filename);
        const fileContent = await fs.readFile(photoPath); // Tauri fs plugin
        zip.file(photo.path, fileContent);
      }
    }
  }

  // Generate ZIP Blob
  const zipBlob = await zip.generateAsync({ type: 'blob' });
  return zipBlob;
};
```

- [ ] ZIP file is created successfully
- [ ] metadata.json is added at root
- [ ] Photos are organized in `/photos/{entry_id}/` structure
- [ ] ZIP size is reasonable (test with ~100 entries + 50 photos)
- [ ] No corruption of photo data

### 4.3: File Picker & Download

**File:** `src/hooks/useExportFile.ts` (new)

Use Tauri's file dialog to let user choose save location:

```typescript
import { save } from '@tauri-apps/plugin-dialog';
import { writeFile } from '@tauri-apps/plugin-fs';

const useExportFile = () => {
  const saveExportZip = async (zipBlob: Blob): Promise<string | null> => {
    // Open file save dialog
    const filePath = await save({
      defaultPath: `chronicle-export-${new Date().toISOString().split('T')[0]}.zip`,
      filters: [
        { name: 'ZIP Archive', extensions: ['zip'] },
        { name: 'All Files', extensions: ['*'] },
      ],
    });

    if (!filePath) return null; // User cancelled

    // Write ZIP to selected location
    const arrayBuffer = await zipBlob.arrayBuffer();
    const bytes = new Uint8Array(arrayBuffer);
    await writeFile(filePath, bytes);

    return filePath;
  };

  return { saveExportZip };
};
```

**Acceptance Criteria:**
- [ ] File dialog opens with default filename (date-stamped)
- [ ] User can choose save location
- [ ] ZIP is written to selected path
- [ ] File size is correct (no truncation)
- [ ] File can be opened with standard ZIP tools

### 4.4: Export Button in Settings

**File:** `src/components/SettingsView.tsx` (extend from Plan 5.3)

Add "Export Data" button in DataSection:

```typescript
// In SettingsView.tsx
const handleExportData = async () => {
  try {
    showLoadingToast('Preparing export...');
    
    const { collectExportData } = useDataExport();
    const exportData = await collectExportData();
    
    showLoadingToast('Creating ZIP file...');
    const zipBlob = await createExportZip({
      data: exportData,
      photoBasePath: await appDataDir(),
    });
    
    showLoadingToast('Saving file...');
    const { saveExportZip } = useExportFile();
    const savedPath = await saveExportZip(zipBlob);
    
    if (savedPath) {
      showSuccessToast(`Exported to ${savedPath}`);
    } else {
      showInfoToast('Export cancelled');
    }
  } catch (error) {
    showErrorToast(`Export failed: ${error.message}`);
    console.error('Export error:', error);
  }
};

return (
  <button onClick={handleExportData} className="primary-button">
    📥 Export Data
  </button>
);
```

**Acceptance Criteria:**

- [ ] Click opens file save dialog
- [ ] Loading toasts show progress
- [ ] Success toast confirms export location
- [ ] Error handling shows user-friendly messages

## Implementation Notes

- **ZIP Library:** `jszip` is cross-platform and works in Tauri. Alternatively, use Tauri's built-in file APIs + a Rust ZIP library via plugin.
- **File Dialog:** Use `@tauri-apps/plugin-dialog` save function. Provides native OS file picker.
- **Async/await:** Export process is async. Use loading toasts to show progress.
- **Error Messages:** User-friendly, not technical jargon.
- **Timestamps:** All dates in ISO 8601 format (UTC). Client converts to local when displaying.
- **Deferred:** Encrypted ZIP (password-protected), scheduled backups, cloud sync — all v2+.
