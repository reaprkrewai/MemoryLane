```typescript
function calculateDayStreak(entries: { created_at: number }[]): number {
  if (entries.length === 0) return 0;
  const uniqueDates = new Set(
    entries.map((e) => format(startOfDay(new Date(e.created_at)), "yyyy-MM-dd"))
  );
  let streak = 0;
  let cursor = startOfDay(new Date());
  while (uniqueDates.has(format(cursor, "yyyy-MM-dd"))) {
    streak += 1;
    cursor = subDays(cursor, 1);
  }
  return streak;
}
// Algorithm preserved in dbQueries.ts but reading local_date strings (TZ-safe per D-07).
```

    ```typescript
    /**
     * SQL aggregate helpers for dashboard widgets.
     * Pagination-independent — returns true DB counts, not allEntries.length.
     * Streak query reads local_date (FOUND-03) for TZ-safe results.
     */

    import { getDb } from "./db";

    export interface EntryStats {
      totalEntries: number;
      totalWords: number;
      thisMonth: number;
      totalTags: number;
      dayStreak: number;
    }

    export async function getEntryStats(): Promise<EntryStats> {
      const db = await getDb();

      // Query 1 — aggregate row covering totals/words/this-month/tag-count.
      // Single round-trip to SQLite; uses scalar subqueries so the engine evaluates each
      // count independently (cheaper than UNION/JOIN tricks at small DB sizes).
      const [agg] = await db.select<Array<{
        total: number;
        words: number;
        this_month: number;
        total_tags: number;
      }>>(`
        SELECT
          (SELECT COUNT(*) FROM entries)                                              AS total,
          (SELECT COALESCE(SUM(word_count), 0) FROM entries)                          AS words,
          (SELECT COUNT(*) FROM entries
             WHERE created_at >= (unixepoch('now', 'start of month') * 1000))         AS this_month,
          (SELECT COUNT(*) FROM tags)                                                 AS total_tags
      `);

      // Query 2 — distinct local_date strings, descending. Bounded at 365 to cap iteration cost.
      // Reads local_date (FOUND-03 column) — never reformat created_at in JS for streak math (D-07).
      const dateRows = await db.select<Array<{ local_date: string }>>(
        "SELECT DISTINCT local_date FROM entries WHERE local_date IS NOT NULL ORDER BY local_date DESC LIMIT 365"
      );
      const dateSet = new Set(dateRows.map((r) => r.local_date));

      // Walk back from "today" in user's local TZ, counting consecutive days that have entries.
      // Algorithm preserved from OverviewView.calculateDayStreak; only the date-source changes.
      // toLocaleDateString("en-CA") returns YYYY-MM-DD in the user's local timezone (D-11 convention).
      let dayStreak = 0;
      const cursor = new Date();
      while (dateSet.has(cursor.toLocaleDateString("en-CA"))) {
        dayStreak += 1;
        cursor.setDate(cursor.getDate() - 1);
      }

      return {
        totalEntries: agg.total,
        totalWords: agg.words,
        thisMonth: agg.this_month,
        totalTags: agg.total_tags,
        dayStreak,
      };
    }
    ```

    **Critical implementation notes:**

    - **Import path is `"./db"` not `"../lib/db"`** — `dbQueries.ts` lives in `src/lib/` alongside `db.ts`, so the relative path is sibling.
    - **No try/catch.** PATTERNS section "Error handling pattern" — `getEntryStats()` is critical for the dashboard. Errors propagate to the caller (Plan 04 entryStore D-05). This matches `entryStore.ts:142-169` `saveContent` style: try/catch only when state cleanup is needed; pure reads bubble up.
    - **Two queries, not one.** D-06 explicitly allows up to 2 — the second (`SELECT DISTINCT local_date`) is unavoidable because streak iteration needs the full set of dates. Combining into one query would force CTE/window-function complexity for no measurable gain.
    - **`unixepoch('now', 'start of month') * 1000`** — multiplies by 1000 because `entries.created_at` is stored as ms (see `db.ts:24` DEFAULT clause `(unixepoch('now') * 1000)`).
    - **`cursor` is `const`** but mutated via `setDate()` — the variable binding doesn't change, only the Date object's internal state. Lint clean.
    - **Why `toLocaleDateString("en-CA")`** — the `"en-CA"` locale formats as `YYYY-MM-DD` (matching `local_date` column shape) in the user's local TZ. D-11 establishes this convention; Plan 04 uses the same call inside `createEntry`.
    - **Loop terminates naturally** — when the cursor walks past a day with no entry, the `while` condition fails and `dayStreak` is returned. Empty DB → `dateSet` is empty → loop body never runs → `dayStreak === 0`. Edge case handled.

    **Do NOT:**
    - Add a `try/catch` block.
    - Import `date-fns` — `toLocaleDateString` + `setDate` are sufficient and avoid pulling another module just for streak math.
    - Add additional named exports beyond `EntryStats` and `getEntryStats` (Phase 8 may add more functions to this file later — keep Phase 7 surface minimal).
    - Cache the result module-level — Plan 04's entryStore is the cache layer. This file is a pure async function.
    - Check `import.meta.env.DEV` for diagnostic logs — production-quality DB helper has no logging by default.
